File Handling CheatSheet!

EOF: End of File

Opening a File:
FILE *file;
file = fopen("filename", "mode");

Closing a File:
fclose(file);

Reading from a File:
char ch;
fscanf(file, "%c", &ch);
    Special: fgetc(file);

Writing to a File:
char ch = "A";
fprintf(file, "%c", ch);
    Special: fputc("A", file);


// EXAMPLE
// WAP that reads data from file "input.txt" and writes to "output.txt"
#include<stdio.h>

int main() {
    FILE *file, *des_file;
    char ch;

    file = fopen("input.txt", "r");
    des_file = fopen("output.txt", "w");

    ch = fgetc(file);
    while(ch != EOF) {
        fputc(ch, des_file);
        ch = fgetc(file);
    }

    printf("Done!");
    fclose(file);

    return 0;
}
